187 bool CircularQueue<T>::IsFull() const

190  // front가 0일 때는, rear가 가장 끝 인덱스 도달하면 full.
191  if (front_ == rear_ - 1 || (front_ == 0) & (rear_ >= size_ - 1)) {
192  return true;
193  }
194  else {
195  return false;
196  }
