260 
261  iterator_ = head_;
262  found = false;
263  moreToSearch = (iterator_ != NULL); //변수 초기화
264 
265  while (moreToSearch && !found) //리스트의 끝이 아니면서 아직 찾지 않았으면 반복한다.
