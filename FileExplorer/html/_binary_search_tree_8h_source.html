409 {
410  // base case : 빈 트리에 도달하여 새 노드 삽입
411  if (tree == nullptr) {
412  Node<T>* new_tree = new Node<T>;
413  new_tree->left = nullptr;
414  new_tree->right = nullptr;
415  new_tree->data = key;

481 template<typename T>
482 Node<T>* BinarySearchTree<T>::RemoveMaxNode(Node<T>* tree, Node<T>* max_node)
483 {
484  // defensive coding
485  if (tree == nullptr) {
486  return NULL;
487  }
488  // 최댓값 노드 찾았으면 교체 작업
489  if (tree == max_node) {

492  // 있다면 상위 노드에 연결됨
493  return max_node->left;

496  tree->right = RemoveMaxNode(tree->right, max_node);

499 // 하위 노드 중 key와 값이 일치하는 노드 제거
500 template<typename T>
501 Node<T>* BinarySearchTree<T>::Remove(Node<T>* tree, T key)
502 {
503  if (tree == nullptr) {
504  return nullptr;
505  }
506  if (tree->data == key) {
507  // 하위 노드가 1개일 때
508  if (tree->left == nullptr) {
509  Node<T>* right_subtree = tree->right;
510  delete tree;
511  // 하위 노드가 없다면 NULL 반환
512  return right_subtree;
513  }
514  if (tree->right == nullptr) {
515  Node<T>* left_subtree = tree->left;
516  delete tree;
517  // 앞의 if에서 left가 null이 아니었으므로 유효한 노드 반환
518  return left_subtree;
519  }
520 
521  // 제거 대상의 왼쪽 서브트리의 최댓값을 제거 대상과 교체하기
522  // 왼쪽 서브트리의 어떤 값보다도 크고, right값보다는 원래 항상 작음
523  Node<T>* max_node = FindMax(tree->left); // 저장
524  max_node->left = RemoveMaxNode(tree->left, max_node); // 제거, 정렬
525  max_node->right = tree->right; // 제거된 key 노드 대체
526  delete tree;
527  return max_node;
528  }
529  else if (key < tree->data) {
530  tree->left = Remove(tree->left, key);
531  }
532  else {
533  tree->right = Remove(tree->right, key);
534  }
535  return tree;
536 }
