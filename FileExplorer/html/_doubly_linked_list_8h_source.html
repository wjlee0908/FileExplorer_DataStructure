271 template <typename T>
272 int DoublyLinkedList<T>::Delete(T item)
273 {

275  bool is_found;
276 
277  // item과 일치하는 노드 탐색해서 일치하는 노드 deleted_node에 할당
278  deleted_node = GetNode(item);
279  is_found = (deleted_node != nullptr);
280 
281  // item과 일치하는 노드 찾으면
282  if (is_found) {
283  if(length_ == 1) { // 하나 남은 노드일 경우 head tail 초기화
284  head_ = nullptr;
285  tail_ = nullptr;
286  }
287  else if (deleted_node == head_) { // head일 경우 head 재설정
288  deleted_node->next->previous = nullptr;
289  head_ = head_->next;
290  }
291  else if(deleted_node == tail_){ // tail일 경우 tail 재설정
292  deleted_node->previous->next = nullptr;
293  tail_ = tail_->previous;
294  }
295  else {

297  deleted_node->previous->next = deleted_node->next;
298  deleted_node->next->previous = deleted_node->previous;
299  }
300

303  length_--;
304  }
